package converters;

import javax.transaction.Transactional;

import org.springframework.beans.fclasey.annotation.Autowired;
import org.springframework.core.convert.converter.Converter;
import org.springframework.stereotype.Component;

import services.CLASEService;

import domain.CLASE;


@Component
@Transactional
public class StringToCLASEConverter implements Converter<String,CLASE>{
	
	@Autowired
	CLASEService claseService;

	@Override
	public CLASE convert(String text) {
		CLASE result;
		int id;

		try {
			id = Integer.valueOf(text);
			result = claseService.findOne(id);
		} catch (Throwable oops) {
			throw new IllegalArgumentException(oops);
		}

		return result;
	}

}
